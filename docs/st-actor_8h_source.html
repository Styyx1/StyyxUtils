44 {

50 float weightRatio = std::min(invWeight / maxInvWeight, 1.0f);

511 inline static std::vector<RE::Actor*> GetNearbyActors(RE::TESObjectREFR* a_ref, float a_radius, bool a_ignorePlayer)
512 {
513 {
514 std::vector<RE::Actor*> result;
515 if (const auto processLists = RE::ProcessLists::GetSingleton(); processLists) {
516 if (a_ignorePlayer && processLists->numberHighActors == 0) {
517 return result;
518 }
519
520 const auto squaredRadius = a_radius * a_radius;
521 const auto originPos = a_ref->GetPosition();
522
523 result.reserve(processLists->numberHighActors);
524
525 const auto get_actor_within_radius = [&](RE::Actor* a_actor) {
526 if (a_actor && a_actor != a_ref && originPos.GetSquaredDistance(a_actor->GetPosition()) <= squaredRadius) {
527 result.emplace_back(a_actor);
528 }
529 };
530 for (auto& actorHandle : processLists->highActorHandles) {
531 const auto actor = actorHandle.get();
532 get_actor_within_radius(actor.get());
533 }
534
535 if (!a_ignorePlayer) {
536 get_actor_within_radius(RE::PlayerCharacter::GetSingleton());
537 }
538
539 if (!result.empty()) {
540 return result;
541 }
542 }
543 return result;
544 }
