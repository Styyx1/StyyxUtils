562inline bool Connect(const char* pluginName, unsigned int a_minVersion = FUCK_API_VERSION)
563{
564 if (!pluginName || pluginName[0] == '\0')
565 {
566 SKSE::log::error("FUCK API Connection failed: You must provide a valid pluginName.");
567 return false;
568 }
569
570 auto handle = GetModuleHandleW(L"FUCK.dll");
571 if (!handle)
572 return false;
573 auto fetcher = (void* (*)())GetProcAddress(handle, "RequestFUCK");
574 if (!fetcher)
575 return false;
576 auto* iface = static_cast<FUCK_Interface*>(fetcher());
577 if (!iface || iface->version < a_minVersion)
578 {
579 SKSE::log::error("FUCK API Version Mismatch: Expected {}, found {}", a_minVersion, iface ? iface->version : 0);
580 return false;
581 }
582
583 GetInterface() = iface;
584
585 // --- Cache the plugin name globally for this DLL ---
586 g_pluginName = pluginName;
587 GetInterface()->LoadTranslation(pluginName);
588
589 SKSE::log::info("Connected to FUCK API version {}", iface->version);
590 return true;
591}
