72 template <class T>
73 static inline void ChangeGameSetting(RE::Setting *setting, const T &value)
74 {
75 if (!setting)
76 {
77 return;
78 }
79
80 switch (setting->GetType())
81 {
82
83 case RE::Setting::Type::kBool:
84 if constexpr (std::is_same_v<T, bool> || std::is_same_v<T, int> || std::is_same_v<T, uint32_t>)
85 {
86 setting->data.b = static_cast<bool>(value);
87 }
88 break;
89
90 case RE::Setting::Type::kFloat:
91 if constexpr (std::is_floating_point_v<T> || std::is_integral_v<T>)
92 {
93 setting->data.f = static_cast<float>(value);
94 }
95 break;
96
97 case RE::Setting::Type::kSignedInteger:
98 if constexpr (std::is_integral_v<T>)
99 {
100 setting->data.i = static_cast<int32_t>(value);
101 }
102 break;
103
104 case RE::Setting::Type::kUnsignedInteger:
105 if constexpr (std::is_integral_v<T>)
106 {
107 setting->data.u = static_cast<std::uint32_t>(value);
108 }
109 break;
110
111 case RE::Setting::Type::kString:
112 if constexpr (std::is_same_v<T, const char *> || std::is_same_v<T, std::string>)
113 {
114 setting->data.s(value);
115 }
116 break;
117
118 default:
119 break;
120 }
121 };

317 template <typename T>
318 static inline T SafelyAdd(T base, T increment) {
319 static_assert(std::is_arithmetic_v<T>, "T must be a number");
320
321 if (increment < 0)
322 return base;
323
324 // so uint8_t and int8_t are covered
325 using ForMath = std::conditional_t < std::is_integral_v<T> && sizeof(T) < sizeof(int), int, T > ;
326 ForMath safe_base = static_cast<ForMath>(base);
327 ForMath safeI_increment = static_cast<ForMath>(increment);
328 ForMath max_value = static_cast<ForMath>(std::numeric_limits<T>::max());
329
330 ForMath sum = safe_base + safeI_increment;
331 if (sum > max_value)
332 sum = max_value;
333
334 return static_cast<T>(sum);
335 }

349 static_assert(std::is_arithmetic_v<T>, "T must be a number");
350
351 if (increment < 0)
352 return base;
353

356 ForMath safe_base = static_cast<ForMath>(base);
357 ForMath safeI_increment = static_cast<ForMath>(increment);
358 ForMath safe_cap = std::min(static_cast<ForMath>(cap), static_cast<ForMath>(std::numeric_limits<T>::max()));
359
360 ForMath sum = safe_base + safeI_increment;
361 if (sum > safe_cap)
